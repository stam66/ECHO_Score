Public DB As MySQLCommunityServer

Public CurrentUsername As String

Public UserGroup As String

Public Property CurrentVideoFolder As FolderItem

' Session.Opening Event
Sub Opening()
  ' Initialize database connection
  DB = New MySQLCommunityServer
  DB.Host = "your_host"
  DB.Port = 3306
  DB.DatabaseName = "echoscore"
  DB.UserName = "your_username"
  DB.Password = "your_password"
  
  Try
    If Not DB.Connect Then
      Raise New DatabaseException("Database connection failed: " + DB.ErrorMessage)
    End If
  Catch e As DatabaseException
    System.DebugLog("Database connection error: " + e.Message)
    ' Show error message to user
    Var errorMsg As String = "Database connection failed. Please contact the system administrator."
    ' You could show this in a dialog or redirect to an error page
  End Try
  
  ' Initialize navigation manager
  Navigation = New WebNavigationManager(Self)
  
  ' Set video folder
  CurrentVideoFolder = SpecialFolder.ApplicationData.Child("CaseVideos")
  If Not CurrentVideoFolder.Exists Then
    CurrentVideoFolder.CreateFolder
  End If
End Sub

' Session.Closed Event
Sub Closed()
  If DB <> Nil And DB.IsConnected Then
    DB.Close
  End If
End Sub

' ServeVideo Method - Returns WebFile for video streaming
Function ServeVideo(filename As String) As WebFile
  If CurrentVideoFolder = Nil Or Not CurrentVideoFolder.Exists Then
    System.DebugLog("Video folder does not exist")
    Return Nil
  End If
  
  Var videoFile As FolderItem = CurrentVideoFolder.Child(filename)
  
  If videoFile = Nil Or Not videoFile.Exists Then
    System.DebugLog("Video file not found: " + filename)
    Return Nil
  End If
  
  Try
    Var wf As New WebFile
    wf.Data = videoFile.Read
    wf.MIMEType = "video/mp4"
    wf.Filename = filename
    wf.ForceDownload = False
    
    Return wf
  Catch e As IOException
    System.DebugLog("Error reading video file: " + e.Message)
    Return Nil
  End Try
End Function
